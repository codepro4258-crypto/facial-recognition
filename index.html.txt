    <script>
        // ============================================
        // PROFESSIONAL FACE RECOGNITION SYSTEM
        // Using face-api.js for real face detection
        // ============================================

        let video, canvas, displaySize;
        let stream = null;
        let isModelLoaded = false;
        let isRecognitionActive = false;
        let detectionInterval = null;
        let recognizedToday = new Set();
        let detectionCounter = 0;
        let currentCamera = null;

        // Data storage
        let registeredPeople = JSON.parse(localStorage.getItem('faceAttendancePeople')) || [];
        let attendanceRecords = JSON.parse(localStorage.getItem('faceAttendanceRecords')) || {};

        // Capture state
        let isCapturing = false;
        let capturedDescriptor = null;

        // Initialize on page load
        window.addEventListener('load', async () => {
            video = document.getElementById('video');
            canvas = document.getElementById('canvas');
            
            await loadModels();
            await loadCameras();
            updateUI();
            showNotification('System initialized successfully!', 'success');
        });

        // Load face-api.js models
        async function loadModels() {
            try {
                const MODEL_URL = 'https://cdn.jsdelivr.net/npm/@vladmandic/face-api/model';
                
                document.querySelector('.loading-text').textContent = 'Loading AI Models...';
                
                await Promise.all([
                    faceapi.nets.tinyFaceDetector.loadFromUri(MODEL_URL),
                    faceapi.nets.faceLandmark68Net.loadFromUri(MODEL_URL),
                    faceapi.nets.faceRecognitionNet.loadFromUri(MODEL_URL),
                    faceapi.nets.faceExpressionNet.loadFromUri(MODEL_URL)
                ]);
                
                isModelLoaded = true;
                document.getElementById('modelStatus').textContent = 'Loaded ‚úì';
                document.getElementById('loadingOverlay').style.display = 'none';
                showNotification('AI Models loaded successfully!', 'success');
            } catch (error) {
                console.error('Error loading models:', error);
                document.querySelector('.loading-text').textContent = 'Error loading models';
                document.getElementById('modelStatus').textContent = 'Error ‚úó';
                showNotification('Failed to load AI models. Please refresh.', 'error');
            }
        }

        // Load available cameras
        async function loadCameras() {
            try {
                const devices = await navigator.mediaDevices.enumerateDevices();
                const videoDevices = devices.filter(device => device.kind === 'videoinput');
                
                const select = document.getElementById('cameraSelect');
                select.innerHTML = '';
                
                if (videoDevices.length === 0) {
                    select.innerHTML = '<option value="">No camera found</option>';
                    return;
                }
                
                videoDevices.forEach((device, index) => {
                    const option = document.createElement('option');
                    option.value = device.deviceId;
                    option.text = device.label || `Camera ${index + 1}`;
                    select.appendChild(option);
                });
                
                currentCamera = videoDevices[0].deviceId;
            } catch (error) {
                console.error('Error loading cameras:', error);
            }
        }

        // Change camera
        async function changeCamera() {
            const select = document.getElementById('cameraSelect');
            currentCamera = select.value;
            
            if (stream) {
                await stopSystem();
                await new Promise(resolve => setTimeout(resolve, 500));
                await startSystem();
            }
        }

        // Start the system
        async function startSystem() {
            if (!isModelLoaded) {
                showNotification('Please wait for AI models to load', 'error');
                return;
            }

            try {
                document.getElementById('detectionStatus').textContent = 'üöÄ Starting camera...';
                
                const constraints = {
                    video: {
                        deviceId: currentCamera ? { exact: currentCamera } : undefined,
                        width: { ideal: 1280 },
                        height: { ideal: 720 },
                        frameRate: { ideal: 30 }
                    },
                    audio: false
                };

                stream = await navigator.mediaDevices.getUserMedia(constraints);
                video.srcObject = stream;
                
                await new Promise(resolve => {
                    video.onloadedmetadata = () => {
                        video.play();
                        resolve();
                    };
                });

                // Set canvas size
                displaySize = { width: video.videoWidth, height: video.videoHeight };
                faceapi.matchDimensions(canvas, displaySize);
                
                document.getElementById('startBtn').disabled = true;
                document.getElementById('recognitionBtn').disabled = false;
                document.getElementById('stopBtn').disabled = false;
                document.getElementById('captureBtn').disabled = false;
                
                const quality = `${video.videoWidth}x${video.videoHeight}`;
                document.getElementById('cameraStatus').textContent = `Active - ${quality}`;
                document.getElementById('detectionStatus').textContent = '‚úÖ Camera active';
                
                showNotification('Camera started successfully!', 'success');
                
                // Start continuous detection
                startDetection();
                
            } catch (error) {
                console.error('Camera error:', error);
                showNotification(`Camera error: ${error.message}`, 'error');
                document.getElementById('detectionStatus').textContent = '‚ùå Camera error';
            }
        }

        // Start face detection
        function startDetection() {
            if (!stream) return;
            
            async function detect() {
                if (!stream) return;
                
                try {
                    const detections = await faceapi
                        .detectAllFaces(video, new faceapi.TinyFaceDetectorOptions())
                        .withFaceLandmarks()
                        .withFaceDescriptors()
                        .withFaceExpressions();
                    
                    const resizedDetections = faceapi.resizeResults(detections, displaySize);
                    
                    // Clear canvas
                    const ctx = canvas.getContext('2d');
                    ctx.clearRect(0, 0, canvas.width, canvas.height);
                    
                    if (detections.length > 0) {
                        detectionCounter++;
                        document.getElementById('detectionCount').textContent = detectionCounter;
                        
                        // Draw detections
                        faceapi.draw.drawDetections(canvas, resizedDetections);
                        faceapi.draw.drawFaceLandmarks(canvas, resizedDetections);
                        
                        const faceCount = detections.length;
                        document.getElementById('detectionStatus').textContent = 
                            `‚úÖ ${faceCount} face${faceCount > 1 ? 's' : ''} detected`;
                        
                        // If recognition is active, try to recognize faces
                        if (isRecognitionActive) {
                            await recognizeFaces(detections);
                        }
                        
                        // If capturing, save the descriptor
                        if (isCapturing && detections.length === 1) {
                            capturedDescriptor = detections[0].descriptor;
                            finishCapture();
                        }
                        
                    } else {
                        document.getElementById('detectionStatus').textContent = 'üîç Looking for faces...';
                    }
                    
                } catch (error) {
                    console.error('Detection error:', error);
                }
                
                requestAnimationFrame(detect);
            }
            
            detect();
        }

        // Toggle recognition
        function toggleRecognition() {
            if (registeredPeople.length === 0) {
                showNotification('No people registered yet. Please register someone first.', 'error');
                return;
            }
            
            isRecognitionActive = !isRecognitionActive;
            const btn = document.getElementById('recognitionBtn');
            
            if (isRecognitionActive) {
                btn.textContent = '‚è∏Ô∏è Pause Recognition';
                btn.style.background = 'linear-gradient(135deg, #f59e0b 0%, #d97706 100%)';
                showNotification('Face recognition activated!', 'success');
            } else {
                btn.textContent = 'üëÅÔ∏è Start Recognition';
                btn.style.background = 'linear-gradient(135deg, #10b981 0%, #059669 100%)';
                showNotification('Face recognition paused', 'info');
            }
        }

        // Recognize faces
        async function recognizeFaces(detections) {
            const today = getToday();
            
            for (const detection of detections) {
                let bestMatch = null;
                let bestDistance = Infinity;
                
                // Compare with all registered people
                for (const person of registeredPeople) {
                    const distance = faceapi.euclideanDistance(detection.descriptor, person.descriptor);
                    
                    if (distance < bestDistance) {
                        bestDistance = distance;
                        bestMatch = person;
                    }
                }
                
                // If match is good enough (threshold: 0.6)
                if (bestMatch && bestDistance < 0.6) {
                    const confidence = Math.round((1 - bestDistance) * 100);
                    
                    // Check if not already marked today
                    const recordKey = `${today}-${bestMatch.id}`;
                    if (!recognizedToday.has(recordKey)) {
                        recognizedToday.add(recordKey);
                        markAttendance(bestMatch, confidence);
                    }
                }
            }
        }

        // Mark attendance
        function markAttendance(person, confidence) {
            const today = getToday();
            const now = new Date();
            const timeString = now.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit', second: '2-digit' });
            
            const record = {
                id: person.id,
                name: person.name,
                time: timeString,
                timestamp: now.toISOString(),
                confidence: confidence
            };
            
            if (!attendanceRecords[today]) {
                attendanceRecords[today] = [];
            }
            
            attendanceRecords[today].push(record);
            saveAttendanceData();
            
            updateAttendanceLog();
            updateStats();
            
            showNotification(`‚úÖ ${person.name} marked present (${confidence}% match)`, 'success');
            
            // Flash the detection status
            document.getElementById('detectionStatus').style.background = 'rgba(16, 185, 129, 0.9)';
            setTimeout(() => {
                document.getElementById('detectionStatus').style.background = 'rgba(0, 0, 0, 0.85)';
            }, 1000);
        }

        // Capture for registration
        function captureForRegistration() {
            const name = document.getElementById('personName').value.trim();
            const id = document.getElementById('personId').value.trim();
            
            if (!name || !id) {
                showNotification('Please enter both name and ID', 'error');
                return;
            }
            
            if (registeredPeople.find(p => p.id === id)) {
                showNotification('This ID is already registered', 'error');
                return;
            }
            
            if (!stream) {
                showNotification('Please start the camera first', 'error');
                switchTab('attendance');
                return;
            }
            
            isCapturing = true;
            capturedDescriptor = null;
            
            document.getElementById('captureBtn').disabled = true;
            document.getElementById('registrationStatus').textContent = 
                'üì∏ Capturing face... Please look directly at the camera';
            
            showNotification('Capturing face... Please look at the camera', 'info');
        }

        // Finish capture
        function finishCapture() {
            isCapturing = false;
            
            if (!capturedDescriptor) {
                showNotification('Failed to capture face. Please try again.', 'error');
                document.getElementById('captureBtn').disabled = false;
                return;
            }
            
            // Create a preview image
            const captureCanvas = document.createElement('canvas');
            captureCanvas.width = 150;
            captureCanvas.height = 150;
            const ctx = captureCanvas.getContext('2d');
            ctx.drawImage(video, 0, 0, captureCanvas.width, captureCanvas.height);
            
            const preview = document.getElementById('capturePreview');
            preview.innerHTML = `<img src="${captureCanvas.toDataURL()}" alt="Captured face">`;
            
            document.getElementById('savePersonBtn').disabled = false;
            document.getElementById('captureBtn').disabled = false;
            document.getElementById('registrationStatus').textContent = 
                '‚úÖ Face captured successfully! Click "Save Person" to register.';
            
            showNotification('Face captured! Now click Save Person.', 'success');
        }

        // Save person
        function savePerson() {
            const name = document.getElementById('personName').value.trim();
            const id = document.getElementById('personId').value.trim();
            
            if (!capturedDescriptor) {
                showNotification('Please capture a face first', 'error');
                return;
            }
            
            const person = {
                id: id,
                name: name,
                descriptor: Array.from(capturedDescriptor),
                registeredDate: new Date().toISOString()
            };
            
            registeredPeople.push(person);
            localStorage.setItem('faceAttendancePeople', JSON.stringify(registeredPeople));
            
            updatePeopleList();
            updateStats();
            resetRegistration();
            
            showNotification(`‚úÖ ${name} registered successfully!`, 'success');
        }

        // Reset registration form
        function resetRegistration() {
            document.getElementById('personName').value = '';
            document.getElementById('personId').value = '';
            document.getElementById('capturePreview').innerHTML = '';
            document.getElementById('savePersonBtn').disabled = true;
            document.getElementById('captureBtn').disabled = false;
            document.getElementById('registrationStatus').textContent = 
                'Ready to capture face. Please ensure good lighting and face the camera directly.';
            
            capturedDescriptor = null;
            isCapturing = false;
        }

        // Stop system
        async function stopSystem() {
            if (stream) {
                stream.getTracks().forEach(track => track.stop());
                stream = null;
            }
            
            isRecognitionActive = false;
            
            document.getElementById('startBtn').disabled = false;
            document.getElementById('recognitionBtn').disabled = true;
            document.getElementById('stopBtn').disabled = true;
            document.getElementById('captureBtn').disabled = true;
            
            const ctx = canvas.getContext('2d');
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            document.getElementById('detectionStatus').textContent = '‚èπÔ∏è System stopped';
            document.getElementById('cameraStatus').textContent = 'Not active';
            
            showNotification('System stopped', 'info');
        }

        // Delete person
        function deletePerson(id) {
            if (confirm('Are you sure you want to delete this person?')) {
                registeredPeople = registeredPeople.filter(p => p.id !== id);
                localStorage.setItem('faceAttendancePeople', JSON.stringify(registeredPeople));
                
                updatePeopleList();
                updateStats();
                
                showNotification('Person deleted successfully', 'success');
            }
        }

        // Update UI functions
        function updatePeopleList() {
            const list = document.getElementById('peopleList');
            
            if (registeredPeople.length === 0) {
                list.innerHTML = '<div class="empty-state">No people registered yet</div>';
                return;
            }
            
            let html = '';
            registeredPeople.forEach(person => {
                const initials = person.name.split(' ').map(n => n[0]).join('').toUpperCase();
                const date = new Date(person.registeredDate).toLocaleDateString();
                
                html += `
                    <div class="person-item">
                        <div class="person-avatar">${initials}</div>
                        <div class="person-info">
                            <h4>${person.name}</h4>
                            <p>ID: ${person.id} ‚Ä¢ Registered: ${date}</p>
                        </div>
                        <button class="delete-btn" onclick="deletePerson('${person.id}')">Delete</button>
                    </div>
                `;
            });
            
            list.innerHTML = html;
        }

        function updateAttendanceLog() {
            const log = document.getElementById('attendanceLog');
            const today = getToday();
            const records = attendanceRecords[today] || [];
            
            if (records.length === 0) {
                log.innerHTML = '<div class="empty-state">No attendance recorded yet today</div>';
                return;
            }
            
            let html = '';
            records.slice().reverse().forEach(record => {
                const confidenceClass = record.confidence >= 85 ? 'high' : 
                                      record.confidence >= 70 ? 'medium' : 'low';
                
                html += `
                    <div class="log-entry new">
                        <strong>${record.time}</strong> - ${record.name}
                        <span class="confidence-badge confidence-${confidenceClass}">
                            ${record.confidence}% match
                        </span>
                        <div class="log-meta">
                            ID: ${record.id}
                        </div>
                    </div>
                `;
            });
            
            log.innerHTML = html;
            
            // Remove 'new' class after animation
            setTimeout(() => {
                document.querySelectorAll('.log-entry.new').forEach(entry => {
                    entry.classList.remove('new');
                });
            }, 2000);
        }

        function updateStats() {
            const today = getToday();
            const records = attendanceRecords[today] || [];
            
            document.getElementById('totalPeople').textContent = registeredPeople.length;
            document.getElementById('presentToday').textContent = records.length;
            document.getElementById('totalRegistered').textContent = registeredPeople.length;
            
            // Calculate total records
            let total = 0;
            Object.values(attendanceRecords).forEach(dayRecords => {
                total += dayRecords.length;
            });
            document.getElementById('totalRecords').textContent = total;
            
            // Calculate storage
            const storage = JSON.stringify(registeredPeople).length + 
                          JSON.stringify(attendanceRecords).length;
            document.getElementById('storageUsed').textContent = Math.round(storage / 1024);
        }

        function updateUI() {
            updatePeopleList();
            updateAttendanceLog();
            updateStats();
            
            const lastBackup = localStorage.getItem('lastBackup');
            if (lastBackup) {
                document.getElementById('lastBackup').textContent = 
                    new Date(lastBackup).toLocaleString();
            }
        }

        // Utility functions
        function getToday() {
            return new Date().toISOString().split('T')[0];
        }

        function saveAttendanceData() {
            localStorage.setItem('faceAttendanceRecords', JSON.stringify(attendanceRecords));
        }

        // Export attendance
        function exportAttendance() {
            const today = getToday();
            const records = attendanceRecords[today] || [];
            
            if (records.length === 0) {
                showNotification('No attendance to export today', 'info');
                return;
            }
            
            let csv = 'Date,Time,Name,ID,Confidence\n';
            records.forEach(record => {
                csv += `${today},${record.time},${record.name},${record.id},${record.confidence}%\n`;
            });
            
            downloadFile(`attendance_${today}.csv`, csv, 'text/csv');
            showNotification('Attendance exported successfully!', 'success');
        }

        // Clear today
        function clearToday() {
            const today = getToday();
            if (confirm('Clear all attendance records for today?')) {
                attendanceRecords[today] = [];
                recognizedToday.clear();
                saveAttendanceData();
                updateAttendanceLog();
                updateStats();
                showNotification('Today\'s attendance cleared', 'success');
            }
        }

        // Backup data
        function backupData() {
            const data = {
                people: registeredPeople,
                attendance: attendanceRecords,
                backupDate: new Date().toISOString()
            };
            
            const json = JSON.stringify(data, null, 2);
            downloadFile(`face_attendance_backup_${getToday()}.json`, json, 'application/json');
            
            localStorage.setItem('lastBackup', new Date().toISOString());
            document.getElementById('lastBackup').textContent = new Date().toLocaleString();
            
            showNotification('Backup created successfully!', 'success');
        }

        // Import data
        function importData() {
            const input = document.createElement('input');
            input.type = 'file';
            input.accept = '.json';
            
            input.onchange = function(e) {
                const file = e.target.files[0];
                const reader = new FileReader();
                
                reader.onload = function(event) {
                    try {
                        const data = JSON.parse(event.target.result);
                        
                        if (data.people) registeredPeople = data.people;
                        if (data.attendance) attendanceRecords = data.attendance;
                        
                        localStorage.setItem('faceAttendancePeople', JSON.stringify(registeredPeople));
                        localStorage.setItem('faceAttendanceRecords', JSON.stringify(attendanceRecords));
                        
                        updateUI();
                        showNotification('Data imported successfully!', 'success');
                    } catch (error) {
                        showNotification('Invalid backup file', 'error');
                    }
                };
                
                reader.readAsText(file);
            };
            
            input.click();
        }

        // Clear all data
        function clearAllData() {
            if (confirm('This will delete ALL registered people and attendance records. Are you sure?')) {
                if (confirm('This action cannot be undone. Continue?')) {
                    registeredPeople = [];
                    attendanceRecords = {};
                    recognizedToday.clear();
                    
                    localStorage.removeItem('faceAttendancePeople');
                    localStorage.removeItem('faceAttendanceRecords');
                    
                    updateUI();
                    showNotification('All data cleared', 'success');
                }
            }
        }

        // Download file helper
        function downloadFile(filename, content, type) {
            const blob = new Blob([content], { type: type });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = filename;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
        }

            event.target.classList.add('active');
            document.getElementById(`${tabName}-tab`).classList.add('active');
        }
